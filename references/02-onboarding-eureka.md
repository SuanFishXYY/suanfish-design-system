const WelcomeOverlay: React.FC = () => {
  const DUR = 2.8;
  const sparks = Array.from({ length: 20 }, (_, i) => {
    const angle = (i / 20) * Math.PI * 2 + (i % 2 === 0 ? 0 : 0.16);
    const dist = 80 + (i % 4) * 35;
    return {
      dx: Math.cos(angle) * dist,
      dy: Math.sin(angle) * dist,
      delay: 0.18 + (i % 8) * 0.025,
      size: 3 + (i % 3),
    };
  });
  const orbs = Array.from({ length: 8 }, (_, i) => {
    const angle = (i / 8) * Math.PI * 2 + 0.4;
    const dist = 160 + (i % 3) * 50;
    return {
      dx: Math.cos(angle) * dist,
      dy: Math.sin(angle) * dist,
      delay: 0.1 + (i % 4) * 0.06,
      size: 28 + (i % 3) * 10,
      warm: i % 2 === 0,
    };
  });
  const rings = [
    { delay: 0.1,  color: 'rgba(251,191,36,0.55)', w: 110 },
    { delay: 0.28, color: 'rgba(244,114,182,0.4)', w: 150 },
    { delay: 0.48, color: 'rgba(251,191,36,0.3)',  w: 190 },
  ];

  return (
    <div
      className="fixed inset-0 z-[60] overflow-hidden pointer-events-none flex items-center justify-center"
      style={{ animation: `onbEurekaBackdrop ${DUR}s ease-out forwards` }}
    >
      {/* 极轻的暖色背景柔光 */}
      <div className="absolute inset-0" style={{
        background: 'radial-gradient(ellipse at center, rgba(255,251,235,0.92) 0%, rgba(255,247,237,0.55) 32%, rgba(255,255,255,0) 72%)',
      }} />

      <div className="relative">
        {/* 3 圈同心扩散环 */}
        {rings.map((r, i) => (
          <span key={`ring-${i}`} className="absolute top-1/2 left-1/2 rounded-full" style={{
            width: r.w, height: r.w, border: `2px solid ${r.color}`,
            boxShadow: `0 0 24px ${r.color}`,
            animation: `onbEurekaRing ${DUR}s cubic-bezier(0.16,1,0.3,1) ${r.delay}s forwards`, opacity: 0,
          }} />
        ))}

        {/* 中心闪点 */}
        <span className="absolute top-1/2 left-1/2 -translate-x-1/2 -translate-y-1/2 rounded-full" style={{
          width: 10, height: 10,
          background: 'radial-gradient(circle, #fff 0%, #fbbf24 40%, transparent 70%)',
          boxShadow: '0 0 50px 14px rgba(251,191,36,0.7)',
          animation: `onbEurekaFlash ${DUR}s cubic-bezier(0.16,1,0.3,1) forwards`,
        }} />

        {/* 8 柔光大光球 */}
        {orbs.map((o, i) => (
          <span key={`orb-${i}`} className="absolute top-1/2 left-1/2 rounded-full -translate-x-1/2 -translate-y-1/2" style={{
            width: o.size, height: o.size,
            background: o.warm
              ? 'radial-gradient(circle, rgba(254,243,199,0.85) 0%, rgba(251,191,36,0.25) 60%, transparent 80%)'
              : 'radial-gradient(circle, rgba(254,226,226,0.8) 0%, rgba(244,114,182,0.22) 60%, transparent 80%)',
            filter: 'blur(6px)',
            '--dx': `${o.dx}px`, '--dy': `${o.dy}px`,
            animation: `onbEurekaOrb ${DUR}s cubic-bezier(0.16,1,0.3,1) ${o.delay}s forwards`, opacity: 0,
          } as React.CSSProperties} />
        ))}

        {/* 20 细小火花 */}
        {sparks.map((s, i) => (
          <span key={`sp-${i}`} className="absolute top-1/2 left-1/2 rounded-full" style={{
            width: s.size, height: s.size,
            background: i % 2 === 0
              ? 'radial-gradient(circle,#fde68a,#f59e0b)'
              : 'radial-gradient(circle,#fecdd3,#f43f5e)',
            boxShadow: i % 2 === 0
              ? '0 0 8px 2px rgba(251,191,36,0.75)'
              : '0 0 8px 2px rgba(244,63,94,0.65)',
            '--dx': `${s.dx}px`, '--dy': `${s.dy}px`,
            animation: `onbEurekaSpark ${DUR}s cubic-bezier(0.16,1,0.3,1) ${s.delay}s forwards`, opacity: 0,
          } as React.CSSProperties} />
        ))}

        {/* 顶部小灯泡气泡 */}
        <div className="absolute left-1/2 -translate-x-1/2 -top-16 flex items-center justify-center" style={{
          animation: `onbEurekaBulb ${DUR}s cubic-bezier(0.16,1,0.3,1) forwards`, opacity: 0,
        }}>
          <span className="relative inline-flex w-10 h-10 rounded-full bg-gradient-to-br from-amber-200 via-amber-300 to-amber-500 items-center justify-center shadow-[0_6px_24px_rgba(245,158,11,0.55)] ring-1 ring-white">
            <SparklesIcon className="w-5 h-5 text-amber-700" />
            <span className="absolute inset-0 rounded-full ring-2 ring-amber-300/60 animate-ping" />
            <span className="absolute -inset-1 rounded-full ring-1 ring-amber-200/50" />
          </span>
        </div>

        {/* 主标题 Welcome <Product>! */}
        <div className="relative text-center px-6">
          <h1
            className="relative text-[3.4rem] md:text-[4.4rem] font-black leading-none tracking-tight overflow-hidden"
            style={{
              fontFamily: '"SF Pro Display","Helvetica Neue",sans-serif',
              animation: `onbEurekaTitle ${DUR}s cubic-bezier(0.16,1,0.3,1) 0.18s forwards`, opacity: 0,
            }}
          >
            <span className="bg-clip-text text-transparent" style={{ backgroundImage: 'linear-gradient(135deg, #475569 0%, #1e293b 60%)' }}>Welcome</span>
            <span className="ml-3 bg-clip-text text-transparent" style={{ backgroundImage: 'linear-gradient(135deg, #f59e0b 0%, #f43f5e 100%)' }}><Product></span>
            <span className="bg-clip-text text-transparent" style={{ backgroundImage: 'linear-gradient(135deg, #f43f5e 0%, #f59e0b 100%)' }}>!</span>
            <span className="absolute inset-y-0 left-0 w-1/3 pointer-events-none" style={{
              background: 'linear-gradient(110deg, transparent 0%, rgba(255,255,255,0.85) 50%, transparent 100%)',
              animation: `onbEurekaShine ${DUR}s cubic-bezier(0.16,1,0.3,1) 0.6s forwards`,
              opacity: 0, mixBlendMode: 'overlay',
            }} />
          </h1>
        </div>
      </div>
    </div>
  );
};
